#include #include #define N 9 // Size of the grid (9x9 for standard Sudoku) bool isSafe(std::vector>& grid, int row, int col, int num) { // Check if 'num' is not present in the current row, column and 3x3 box for (int x = 0; x < N; x++) { // Check row and column if (grid[row][x] == num || grid[x][col] == num) return false; } // Check 3x3 subgrid int startRow = row - row % 3, startCol = col - col % 3; for (int i = 0; i < 3; i++) for (int j = 0; j < 3; j++) if (grid[i + startRow][j + startCol] == num) return false; return true; } bool solveSudoku(std::vector>& grid, int row, int col) { // If we have reached the 9th row, the puzzle is solved if (row == N - 1 && col == N) return true; // Move to the next row if we reach the end of a row if (col == N) { row++; col = 0; } // Skip cells already filled if (grid[row][col] != 0) return solveSudoku(grid, row, col + 1); // Try placing numbers 1 to 9 in the cell for (int num = 1; num <= N; num++) { if (isSafe(grid, row, col, num)) { grid[row][col] = num; // Recursively solve for next cells if (solveSudoku(grid, row, col + 1)) return true; // Backtrack if placing num does not lead to a solution grid[row][col] = 0; } } return false; // Trigger backtracking } void printGrid(const std::vector>& grid) { for (int r = 0; r < N; r++) { for (int d = 0; d < N; d++) { std::cout << grid[r][d] << " "; } std::cout << std::endl; } } int main() { // 0 represents empty cells std::vector> grid = { {5, 3, 0, 0, 7, 0, 0, 0, 0}, {6, 0, 0, 1, 9, 5, 0, 0, 0}, {0, 9, 8, 0, 0, 0, 0, 6, 0}, {8, 0, 0, 0, 6, 0, 0, 0, 3}, {4, 0, 0, 8, 0, 3, 0, 0, 1}, {7, 0, 0, 0, 2, 0, 0, 0, 6}, {0, 6, 0, 0, 0, 0, 2, 8, 0}, {0, 0, 0, 4, 1, 9, 0, 0, 5}, {0, 0, 0, 0, 8, 0, 0, 7, 9} }; if (solveSudoku(grid, 0, 0)) printGrid(grid); else std::cout << "No solution exists." << std::endl; return 0; } Â
